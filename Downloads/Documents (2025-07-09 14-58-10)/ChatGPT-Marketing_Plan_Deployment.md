const { Client, Collection, GatewayIntentBits } = require("discord.js");

const client = new Client({ intents: [GatewayIntentBits.Guilds, GatewayIntentBits.GuildMessages, GatewayIntentBits.MessageContent, GatewayIntentBits.DirectMessages] });

import React, { useState, useEffect } from "react";

export default function DiscordOpsDashboard() {
  const [command, setCommand] = useState("");
  const [param, setParam] = useState("");
  const [logs, setLogs] = useState([]);

  useEffect(() => {
    fetch("/api/logs")
      .then((res) => res.json())
      .then((data) => setLogs(data))
      .catch(() => setLogs([]));

        body: JSON.stringify({ command, param }),
      });
      const data = await res.json();
      setStatus(data.message || "Sent!");
    } catch {
      setStatus("Failed to send command.");
    }
  };

  return (
    <main className="min-h-screen bg-neutral-950 text-neutral-100 p-8 flex flex-col items-center">

            </select>
          </label>
          {(command === "ship" || command === "ask") && (
            <label className="flex flex-col gap-1">
              <span className="font-medium">
                {command === "ship" ? "Project Name" : "Prompt"}
              </span>

                type="text"
                value={param}
